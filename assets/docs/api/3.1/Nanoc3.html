def self.load_plugins
  # Don’t load if there’s no rubygems
  return false if !defined?(Gem)

  Gem.source_index.find_name('').each do |gem|
    # Skip irrelevant ones
    next if gem.name !~ /^nanoc3-/

    # Load plugin
    require gem.name
  end

  true
end
