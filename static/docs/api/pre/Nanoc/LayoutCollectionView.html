def [](arg)
  layout = find { |l| l.identifier == arg }
  return layout if layout

  # FIXME: this should only work if globs are enabled
  pat = Nanoc::Int::Pattern.from(arg)
  layout = find { |l| pat.match?(l.identifier) }
  return layout if layout

  nil
end
