def [](*args)
  if 1 == args.size && args.first.is_a?(String)
    item_with_identifier(args.first) || item_matching_glob(args.first)
  elsif 1 == args.size && args.first.is_a?(Regexp)
    @objects.select { |i| i.identifier.to_s =~ args.first }
  else
    @objects[*args]
  end
end

def find_all(arg)
  pat = Nanoc::Int::Pattern.from(arg)
  @objects.select { |o| pat.match?(o.identifier) }
end
