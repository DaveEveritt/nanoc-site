def initialize(hash)
  @hash             = hash
  @unprocessed_keys = @hash.keys.dup
  @mutex            = Mutex.new
end

def next_pair
  @mutex.synchronize do
    key = @unprocessed_keys.shift
    return (key ? [ key, @hash[key] ] : nil)
  end
end
