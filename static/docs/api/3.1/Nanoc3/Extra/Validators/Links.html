def run
  require 'nokogiri'

  @delegate = self
  links = all_broken_hrefs
  if links.empty?
    puts "No broken links found!"
  else
    links.each_pair do |href, origins|
      puts "Broken link: #{href} -- referenced from:"
      origins.each do |origin|
        puts "    #{origin}"
      end
      puts
    end
  end
end
