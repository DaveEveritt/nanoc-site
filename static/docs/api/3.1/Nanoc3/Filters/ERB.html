def run(content, params={})
  require 'erb'

  # Create context
  context = ::Nanoc3::Context.new(assigns)

  # Get result
  erb = ::ERB.new(content)
  erb.filename = filename
  erb.result(context.get_binding { assigns[:content] })
end
