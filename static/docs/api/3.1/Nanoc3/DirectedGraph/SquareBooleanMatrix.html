Overview

Nanoc3::DirectedGraph::SquareBooleanMatrix is, as the name says, a square matrix that contains boolean values. It is used as an adjacency matrix by the Nanoc3::DirectedGraph class.

def to_s
  s = ''

  # Calculate column width
  width = (@size-1).to_s.size

  # Add header
  s << ' ' + ' '*width + ' '
  @size.times { |i| s << '| ' + format("%#{width}i", i) + ' ' }
  s << "\n"

  # Add rows
  @size.times do |x|
    # Add line
    s << '-' + '-'*width + '-'
    @size.times { |i| s << '+-' + '-'*width + '-' }
    s << "\n"

    # Add actual row
    s << ' ' + format("%#{width}i", x)+ ' '
    @size.times do |y|
      s << '| ' + format("%#{width}s", self[x, y] ? '*' : ' ') + ' '
    end
    s << "\n"
  end

  # Done
  s
end
